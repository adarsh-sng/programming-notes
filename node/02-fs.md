```js
const fileContents=fs.readFileSync("../file.txt")// returns a buffer
console.log(fileContents.toString()) // convert buffer to string
const fileContents=fs.readFileSync("../file.txt", "utf-8") // read file synchronously
console.log(fileContents)
```

```js
fs.readFile("../file.txt","utf-8",(err,data)=>{
  if(err){
    console.error(err)
    return
  }
  console.log(data)
})
```

```js 
fs.writeFileSync("../file.txt","Hello World") // write file synchronously
fs.writeFile("../file.txt","Hello World",(err)=>{
  if(err){
    console.error(err)
    return  
  }else{
    console.log("File written successfully")
  }
})
```
```js
fs.writeFile("../file.txt","Hello World",{flag:"a"},(err)=>{ // append to file
  if(err){
    console.error(err)
    return  
  }else{
    console.log("File written successfully")
  }
})

## fs promises module
``` js
const fs = require("fs/promises")

fs.readFile("../file.txt", "utf-8")
  .then((data) => {
    console.log(data)
  })
  .catch((err) => {
    console.error(err)
  })
```
